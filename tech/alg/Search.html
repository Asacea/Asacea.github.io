    • 折半查找法

      int BinSearch(RecordList l,KeyType k){
    int low=1;
    int high=l.length;
    int mid;
    while(low<=high){
        mid=(low+high)/2;
        if(k==l.r[mid].key)
            return mid;
        else if(k<l.r[mid].key)
            high=mid-1;
        else
            low=mid+1;
    }
    return 0;
}

  1. 二叉排序树

    • 定义：左子树上所有结点的值均小于根结点的值；右子树上的所有结点的值均大于根结点

    • 数据类型

    • 二叉排序树的查找

      //递归
BSTree SearchBST(BSTree bst,KeyType key){
    if(bst==NULL)  return NULL;
    else if(bst->key==key)  return bst;
    else if (bst->key>key)
        return SearchBST(bst->LChild,key);
    else
        return SearchBST(bst->RChild,key);
}
//非递归
BSTree SearchBST(BSTree bst,KeyType key){
    BSTree q;
    q=bst;
    while(q){
        if(q->key==key) return q;
        if(q->key>key) 
            q=q->LChild;
        else
            q=q->RChild;
    }
    return NULL;
}

    • 二叉排序树的删除

      BSTNode *DelBST(BSTree t,KeyType k){
    BSTNode *p,*f,*s,*q;//p是要删除的结点，f是p的双亲结点，s是p结点在中序遍历中的直接前驱，q是s的双亲结点，S就是p的左子树的最右结点
    p=t;
    f=NULL;
    //找到p结点，定位f结点
    while(p){
        if(p->key==k)  break;
        f=p;
        if(p->key>k)   p=p->LChild;
        else  p=p->RChild;
    }
    if(p==NULL)  return t;//若找不到，返回原来的二叉排序树
    
    if(p->LChild==NULL){ //如果p没有左子树，直接将p的右子树连到f上
        if(f==NULL)  //如果p是根结点
            t=p->RChild;//直接将树根指向p的右孩子
        else if(f->LChild==p)//如果p是f的左孩子
            f->LChild=p->RChild;
        else //如果p是f的右孩子
            f->RChild=p->RChild;
        free(p);
    }
    else{ //p有左子树
        q=p;//提示：q是s的双亲结点
        s=p->LChild;
        //找到p的中序前驱结点s，即p的左子树的最右下结点，因此s没有右孩子
        while(s->RChild){
            q=s;//q是s的双亲结点
            s=s->RChild;
        }
        //因为s没有右孩子，所以只用考虑左子树
        if(q==p)//如果p是s的双亲结点，s是p的左孩子
            q->LChild=s->LChild;//将原s的左子树接到q的左孩子
        else 
            q->RChild=s->LChild;//将原s的左子树接到q的右孩子
        p->key=s->key;
        free(s);//删除s结点
    }
    return t;
}

  2. 平衡二叉排序树

    • 定义：1.左右子树的高度之差的绝对值小于1（1、0、-1）；2.左右子树也是平衡二叉排序树

    • 平衡因子：该结点的左右子树的高度之差

    • 平衡二叉树的调整：一般下层的祖先结点先恢复平衡，将使上层的祖先结点恢复平衡，因此应该调整最下面的失衡子树

    • LL型

      //LL型：在左子树的左子树插入结点导致失衡
//失衡特点：A->bf==2,B->bf==1
//方法：将B的右孩子改为A的左孩子，将A改为B的右孩子
A->LChild=B->RChild;
B->RChild=A;
A->bf=0;
B->bf=0;
//设A原来的双亲结点为FA，将B接到FA
if(FA==NULL)//若A为根结点
    t=B;
else if(A==FA->LChild)
    FA->LChild=B;
else if(A==FA->RChild)
    FA->RChild=B;

    • LR型

      //失衡点的左子树的右子树插入结点导致失衡
//失衡特点：A->bf==2,B->bf==-1
//方法：将C的右孩子改为A的左孩子，将A改为C的右孩子，将C的左孩子改成B的右孩子，将B改成C的左子树
B->RChild=C->LChild;
A->LChild=C->RChild;
C->Lchild=B;
C->RChild=A;
//修改平衡因子
if(S->key<C->key){ //如果S是插在C的左子树
    A->bf=-1;B->bf=0;C->bf=0;
}
else if(S->key<C->key){ //如果S是插在C的右子树
    A->bf=0;B->bf=1;C->bf=0;
}
else if(S->key=C->key){ //如果C结点本身就是插入的新结点S，即原来B的右子树为空，Cl、Cr、Bl、Ar均为空
    A->bf=0；B->bf=0;
}
//修改根结点,FA是A的双亲结点，t指向根结点
if(FA=NULL)  t=C;
else if(A==FA->LChild)  FA->LChild=C;
else FA->RChild=C;

    • RR型

//RR型的失衡特点是：A->bf=-2,B->bf=-1;

//与LR型对称,失衡特点：A->bf==-2,B->bf==1
//方法：将C的左子树改成A的右子树，将C的右子树改成B的左子树，将A改成C的左子树，将B改成C的右子树

    A->bf=0;B->bf=0;
}
//修改根指针
if(FA=NULL)  t=C;
else if(A==FA->LChild)  FA->LChild=C;
else FA->RChild=C;

    • 完整插入代码

      //算法思想：
//1.寻找插入位置，并且找到失衡结点A
//2.插入结点S
//3.确定结点B,B是
//4.修改从B到S路径上的个结点的平衡因子
//5.根据前面的四种方法，来调整失衡树
typedef struct node{
    KeyType key;
    int bf;
    struct node *RChild,*LChild;   
}ALVTNode,*AVLTree;

void ins_AVLTree(AVLTree *avlt,KeyType K){
    //创建新结点S
    AVLTNode *S,*A,*FA,*p,*fp;
    S=(AVLTNode*)malloc(sizeof(AvlTNode));
    S->key=k;
    S->LChild=S->RChild=NULL;
    S->bf=0;
    if(*avlt==NULL)  *avlt=S;
    else{
        //首先找到S的插入位置，同时记录距离S的插入位置最近且平衡因子不为0的结点A，A 可能 是失衡结点
        A=*avlt;FA=NULL;
        p=*avlt;fp==NULL;
        while(p!=NULL){ //寻找p的位置
            if(p->bf!=0){
                A=p;FA=fp;
            }
            fp=p;
            if(k<p->key)  p=p->LChild;
            else p=p->RChild;
        }
        //此时p=NULL
        //插入S
        if(k<fp->key)    fp->LChild=S;
        else   fp->RChild=S;
        //确定结点B（B一般是A的S所在子树的祖先节点），并修改A的平衡因子
        if(k<A->key){
            B=A->LChild;A->bf+=1;
        }else{
            B=A->RChild;A->bf-=1;
        }
        //修改B到S的路径上各结点的平衡因子（原值均为0）
        p=B;
        while(p!=S){
            if(k<p->key){
                p->bf=1;p=p->LChild;
            }else{
                p->bf=-1;p=p->RChild;
            }
        }
        //判断失衡类型并做相应处理
        if(A->bf==2&&B->bf=1)
        {/*LL型*/}
        else if(A->bf==2&&B->bf=-1)
        {/*LR型*/}
        else if(A->bf=-2&&B->bf==-1)
        {/*RR型*/}
        else if(A->bf==-2&&B->bf==1)
        {/*RL型*/}
    }
}

# 计算式查找法——哈希法

基本思想：在元素关键词k和元素的存储位置p之间建立一种函数关系

冲突问题：k1≠k2,但是 H(k1)=H(k2),则称k1和k2 同义

# 哈希函数的构造方法

  1. 直接地址法

    关键词的某个线性函数作为哈希地址：Hash(key)=a*key+b

  2. 数字分析法

    • 构造方法：在关键词比较长且形式已知的情况下，选取随机性好的若干位作为哈希地址

  3. 平方取中法

    • 构造方法：取关键词平方值的中间几位作为哈希地址
    • Hash(key)="key的平方的中间几位"
    • 扩大差别，贡献均匀；适用于不知道全部关键词的情况

  4. 分段叠加法

    • 构造方法：将关键词按照哈希表地址位数（eg:哈希表长度为1000，则三位一段），分成位数相等的几部分（不足的舍去）；然后将这几部分叠加和作为哈希地址
    • 移位法：直接相加
    • 折叠法：隔一个数就把一个数倒过来相加P301

  5. 除留余数法

    • 构造方法：Hash(key)=key %p,p<=m(通常选择一个不大于m的素数)

  6. 伪随机数法

    • 构造方法：Hash(key)=random(key)

# 处理冲突的方法

  1. 开放定址法

    • 当出现冲突的时候,以冲突的地址为关键词寻找下一个哈希地址

    • 线性探测法

    • 平方探查法

  2. 双散列法（再哈希法）

    • Hi=RHi(key),i=1,2,3,...,k
    • 不容易产生聚集，但是增加了计算时间

  3. 连地址法

    • 基本思想L：将所有哈希地址为i 的元素构成一个单链表中（称为同义词链），并将单链表的头指针存储在哈希表的的第i个单元中

  4. *哈希算法的性能分析

  • 装填因子：a=表中填入的记录数/哈希表的长度

# 哈希表的查找过程

#define m 100  //哈希表的长度
#define NULLKEY  100//代表空记录的关键词值
typedef int KeyType;
typedef struct {
    KeyType key;
}RecordType;
typedef RecordType HashTable[m];

int HashSearch(HashType ht,KryTypr k){
    int h0=Hash(k);
    if(ht[h0].key==NULLKEY)  return -1;
    else if(ht[h0].key==k)  return h0;
    else{
        for(int i=0;i<m;i++){
            hi=(h0+i)%m;
            if(ht[hi].key==NULLKEY)  return -1;
            else if(ht[hi].key==k)   return hi;
        }
        return -1;
    }        
}
