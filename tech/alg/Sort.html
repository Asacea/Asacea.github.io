# 直接插入排序（略）

类似于数组的排序,按照关键词排序的规则找到插入位置后，将后面的元素都向移动一位

时间复杂度：O(n^2)

//设置监视哨r[0],作用是1.防止越界；2.暂存需插入的数据
void InsSort(RecordType r[],int length){
    int i,j;
    for(i=2;i<=length;i++){ //从i=2开始，因为第一个数据自动看成一个有序子数列
        r[0]=r[i];  //将待插入的数据放入监视哨r[0]中
        j=i-1;  //j指向已经排号的有序数列的最后一位
        while(r[0].key<r[j].key){   //寻找插入位置
            r[j+1]=r[j];
            j=j-1;
        }
        r[j+1]=r[0];
    }
}

# 折半插入排序法

时间复杂度：O(n^2)

void BinSort(RecordType r[],int length){
    int i,j;
    int x;//插入数据
    int low,high,mid;
    for(i=0;i<=length;i++){
        x=r[i];
        low=1;
        high=i-1;
        while(low<=high){   //寻找插入位置
            mid=(low+high)/2;
            if(x.key<r[mid].key)
                high=mid-1;
            else
                low=mid+1;
        }
        for(j=i-1;j>=low;--j)   //将插入位置以后的元素后移
            r[j+1]=r[j];
        r[low]=x;
    }
}

# 希尔排序

  1. 适用范围：关键词序列基本有序且个数较少
  2. 方法：将待排序的关键词分为若干个较小的子序列，对子序列进行直接插入排序
  3. 时间复杂度：O(n^1.5)
void ShellInsert(RecordType r[],int length;int delta){  //delta是增量
    int i,j;
    for(i=1+delta;i<length;i++)
        if(r[i].key<r[i-delta].key){
            r[0]=r[i];
            for(j=i-delta;j>0&&r[0].key<r[j].key;j-=delta)
                r[j+delta]=r[j];
            r[delta+j]=r[0];
        }
}
void ShellSort(RecordType r[],int length;int delta[],int n){
    for(int i=0;i<n;i++)
        ShellInsert(r,length,delta[i]); 
}

void bubbleSort(int a[],int n){
    int i,j,x;
    for(j=0;j<n-i;j++)
        for(i=n-2;i>=j;i--){
            if(a[i]>a[i+1]){
                x=a[i];
                a[i]=a[i+1];
                a[i+1]=x;
            }
        }
}
void BubbleSort(int a[],int n){
    int i,j,x;
    int flag=1;
    j=n-2;//j是本遍扫描的终点下标，初始值为n-2
    while(flag){
        flag=0;
        for(i=0;i<=j;i++){
            if(a[i]>a[i+1]){
                x=a[i];
                a[i]=a[i+1];
                a[i+1]=x;
                flag=1;
            }
        }  
        j--;
    }
}

# 快速排序

思路P329,时间复杂度：O(n^2)

void QkSort(RecordType r[],int low,int high){
    int pos;
    if(low<high){
        pos=QKPass(r,low,high);
        QKSort(r,low,pos-1);
        QKSort(r,pos+1,high);
    }
}
int QKPass(RecordType r[],int low,int high){
    int x=r[low];
    while(low<high){
        while(low<high&&r[high].key>=x.key)
            high--;
        if(low<high){
            r[low]=r[high];
            low++;
        }
        while(low<high&&r[low].key<x.key)
            low++;
        if(low<high){
            r[high]=r[low];
            high--;
        }
    }
    r[low]=x;
    return  low;
}

void SelectSort(RecordType r[],int n){
    for(int i=1;i<=n-1;i++){
        k=i;
        for(int j=i+1;j<=n;j++)//从i以后的无需序列中选出关键词最小的记录k
            if(r[j].key<r[k].key)
                k=j;
        if(k!=i){
            x=r[i];
            r[i]=r[k];
            r[k]=x;
        }
    }
}

# 堆排序

  1. 定义：排序过程中将向量中存储的数据看成一颗完全二叉树，任意结点r[i]的左孩子是r[2i]，右孩子是r[2i+1]

  2. 堆定义：完全二叉树满足条件：r[i].key>=r[2i].key&&r[i].key>=r[2i+1].key——大根堆

    r[i].key<=r[2i].key&&r[i].key<=r[2i+1].key——小根堆

  3. 重建堆

    将根顶记录改变后，从上到下调整堆

    void sift(RecordType r[],int k,int m){   //设，r[k.m]是以r[k]为根的完全二叉树，大根堆,m是堆中最后一个记录

# 归并排序法

  1. 基本思想：将两个或者两个以上的有序表合并成一个新的有序表

  2. 空间复杂度

  3. //已知r1[low..mid]和r1[mid+1..high]分别按关键词有序排列，将它们合成存放在r2[low..high]中
void Merge(RecordType r1[],int low,int mid,int high,RecordType r[]){
   int i,j,k;
   i=low;
   j=mid+1;
   k=low;
   while(i<=mid&&j<=high){
       if(r1[i].key<=r1[j].key){
           r[k]=r1[i];
           i++;
       }else{
           r[k]=r1[j];
           j++;
       }
       k++;
   }
   while(i<=mid){
       r[k]=r1[i];
       k++;
       i++;
   }
   while(j<=high){
       r[k]=r1[j];
       k++;
       j++;
   }
}

    void MergeSort(RecordType r1[],int low,int high,RecordType r[]){
    RecordType *r2;
    r2=(RecordType*)malloc(sizeof(RecordType)*(high-low+1));
    if(low==high)
        r[low]=r1[low];
    else{
        mid=(low+high)/2;
        MergeSort(r1,low,mid,r2);
        MergeSort(r1,mid+1,high,r2);
        Merge(r2,low,mid,high,r);
    }
    free(r2);
}

f[j]和e[j]分别指向队列中的第一个和最后一个，f[j]=0表示是空队列*/
void Distribute(RecordType r[],int i,PVector f,PVector e){
    int j;
    for(j=0;j<=RADIX-1;j++)
        f[i]=0;
    int p=r[0].next;//p指向链表中的第一个记录
    while(p!=0){
        j=Order(r[p].key[i]);
        if(f[j]==0) 
            f[j]=p;
        else
            r[e[j]].next=p;
        e[j]=p;
        p=r[p].next;
    }
}
void Collect(RecordType r[],PVector f,PVector e){
    int j=0;
    while(f[j]==0) j++;//找到第一个非空队列
    r[0].next=f[j];
    t=e[j];
    while(j<RADIX-1){
        j++;
        while((j<RADIX-1)&&(f[j]==0))   j++;//找下一个非空队列；
        if(f[j]!=0){
            r[t].next=f[j];
            t=e[j];
        }
    }
    r[t].next=0;
}
