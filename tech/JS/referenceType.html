# Array

  1. 初始化

    let arr = new Array(1,2,3)
let arr =[1,2,3]

  2. 相关基础方法

    /*Array.from() 把不是数组的东西转化成真数组，就可以使用数组的特性
  例如：string，arguements，document.querySelectorAll("li")*/
let arr = Array.from('abc')//['a','b','c']
let arr1 = Array.from(arr,(x)=>x+"a")//["aa","ba","ca"]
let arr0 = Array.of(3)//创建只有一个元素的数组，[3]
//数组空位
let arr = [1,,3,,5]

//迭代对象
let arr=['a','b',3,4]
arr.keys()//[0,1,2,3]
arr.values()//['a','b',3,4]
arr.entries()//[[0,'a'],[1,'b'],[2,3],[3,4]]
for(let [index,ele] of arr.entries()){
    console.log(index,ele)
}

//扁平化
let arr4 = [1,2,3,[1,2]]
let arr5 =  arr4.flat() //flatMap()用于数组的元素是Object的时候，()中是一个函数，返回需要扁平化的对象

//测试
let isTrue=arr.every((ele)=>ele>3)//判断是否所有的元素都>3
let arr3=arr.fliter((ele)=>ele>3)//取出所有>3的元素
arr.forEach((ele，index)=>console.log(ele,index))//遍历
let arrmap=arr.map((ele)=>ele+2)//所有元素的操作

//增删数组元素
let arr = new Array().fill(0)
arr.push(0)
arr.pop()//栈
arr.shift()//队列，取第一项
arr.concat([5,6])
arr.slice(startIndex,endIndex)

//排序
arr.reserve()
arr.sort((a,b)=>a-b)//注意：数组的sort方式是按照字符串的顺序比较的，如果是数字排序的话，要写成函数

//查找元素
arr.indexOf('b')//1
arr.includes('a')//true
arr.find((ele,idx,array)=>typeof(ele)==Number)

//归并
let arr=[1,2,3,4]
let sum=arr.reduce((prev,cur)=>prev+cur)//10

# Map

  1. Map可以使用任何的数据类型作为键，不只局限于字符串（对比Object）。

    Map与Object的区别：

  2. 基本方法:

    //初始化
const map = new Map();
const map1 = new Map([[key1,val1],[key2,val2]]);//二维数组
//增删查改
map.set(key3.val3);
map.delete(key3);//删除键值对

map.get(key3);//获取val3

  3. 迭代方法

    //映射
for(let [val,key] of map.entries()){	
    console.log(val,key);
}
for(let key of map.keys()){
    console.log(key)
}
for(let val of map.values()){
    console.log(val)
}
//forEach
map.forEach((val,key)=>alert(`${key} ->${val}`))//字面量模板
//将map转化成数组——内置迭代器
let m = [...map];

    注意：键和值在迭代器遍历时时可以修改的，但是不可以在映射内部修改，但是如果键或者值是object的时候，可以修改它的内部属性

# Set

  1. 与数组的区别是，Set 中的元素不允许出现重复，array则可以出现重复。Set中也允许任何数据类型。数组中有index，set没有！！！

  2. 基本方法

    //初始化
let s1 = new Set(1,2,3,4,4,4,4)//自动转化成1，2，3，4去重
console.log([...s1]);//[1,2,3,4],所以这个方法可以完成数组去重
//例如
let list = [1,2,3,21,2,[1,3],[1,3],{a:1},{a:1},"lily","lily"]
function uni(arr){
    let res = new Set()
    return arr.filter((item)=>{
        let id=JSON.stringify(item)//唯一性判断
        if(res.has(id)){
            return false
        }else{
            res.add(id)
            return true
        }
    })
}
console.log(uni(list))
