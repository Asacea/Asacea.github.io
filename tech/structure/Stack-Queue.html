# 顺序栈

  1. 存储结构

    #define Stack_Size 50
typedef struct{
    StackElementType elem[Stack_Size];
    int top;   //存放栈顶元素的下标，top=-1为空栈
}SeqStack;

  2. 基本操作：

    #define TRUE 1;

    #define FALSE 0;

    • 初始化：

      void InitStack(SeqStack *S)
{
    S->top=-1;
}

    • 判断

      int isFull(SeqStack *S){
    return S->top==Stack_Size-1 ? TRUE : FALSE;
}
bool isEmpty(SeqStack *S){
    return S->top==-1 ? TRUE : FALSE; 
}

    • 进栈和出栈

      int Push(SeqStack *S,StackElementType x)
{
    if(isFull(S))
        return FALSE;
    S->top++;
    S->elem[S->top]=x;
    return TRUE;
}

int Pop(SeqStack *S,StackElementType *x)
{
    if(isEmpty(S))
            return FALSE;
    else
    {
        *x=S->elem[S->top];
        S->top--;
        return TRUE;
    }
}

    • 两栈共享技术

      #define M 100
typedef struct 
{
    StackElementType Stack[M];
    StackElementType top[2];
}DqStack;
void InitStack(DqStack *S)
{
    S->top[0]=-1;
    S->top[1]=M;   //表示空栈
}

# 链栈

  1. 结构:

    typedef struct node
{
    StackElementType data;
    struct node *next;
}LinkStackNode,*LinkStack;
    • 始终有一个栈顶指针top,指向当前栈顶元素的头结点
    • 若top->next==NULL，则栈空
    • 链栈不会溢出

  2. 栈操作

    int Push(LinkStack top,StackElementType x)
{
    //头插入
}

    int Pop(LinkStack top,StackElementType *x)//取头元素
{
    LinkStackNode *temp;
    temp=top->next;
    if(temp==NULL)
        return FALSE;
    top->next=temp->next;
    *x=temp->data;
    free(temp);
    return TRUE;
}

    int GetTop(LinkStack S,StackElementType *x)
{
    if(S->top==-1)
        return FALSE;
    else
    {
        *x=S->elem[S->top];
        return TRUE;
    }
}

# 表达式求值(无括号)

  • 考虑因素：
  1. 从左至右依次读取数和运算符
  2. 运算符的优先级
  3. 特殊运算符：^为幂运算，#是表达式结束符

  • 算法逻辑：

    1. 规定运算符的优先级表
    2. 设置两个栈：OVS运算数栈，OPTR运算符栈
    3. 从左至右读入，
      • 遇到运算数：进入运算数栈
      • 遇到运算符，比较优先级
        1. 当前运算符>OPTR栈顶运算符，则当前运算符进栈
        2. 当前运算符<=OPTR栈顶运算符，则OPTR出栈一次，得到运算符$；OVS出栈两次，得到运算数a,b；进行运算后得到数t,将t入栈

  • 图解：

  int ExpEvaluation()
  {
      SeqStack OPTR,OVS;
      char n,a,b,v;
      char ch,op;
      InitStack(&OPTR);     //1
      InitStack(&OVS);
      Push(&OPTR,'#');     //push(OPTR)2
      printf("input an expression(Ending with =):");
      ch=getchar();
      while(ch!='#'||GetTop(&OPTR)!='#')  //3
      {
          if(!In(ch,OPSet))   //4
          {
              n=GetNumber(ch);   //5
              Push(&OVS,n);     //6
              ch=getchar();
          }
          else
              switch(Compare(ch,GetTop(&OPTR)))  //7
              {
                  case  '>':
                      Push(&OPTR,ch);
                      ch=getchar();
                      break;
                  case '=':
                  case '<':
                      Pop(&OPTR,&op);        //8
                      Pop(&OVS,&a);
                      Pop(&OVS,&b);
                      v=Execute(a,op,b);        //9
                      Push(&OVS,v);
                      break;
              }
      }
      v=GetTop(&OVS);
      return v;
  }

注意c语言里面的int和char 类型的相似性

# 循环队列

  1. 要点：

    1. 用数组存储数据，并且将他看作是一个环状的空间
    2. 规定最后一个单元的后继为第一个单元，即Queue[MAXSIZE-1]的下一个是Queue[0]
    3. 实现方法：尾指针rear=(rear+1)%MAXSIZE,头指针front=(front+1)%MAXSIZE

  2. 储存方式

    #define MAXSIZE 10
typedef struct
{
    QueueElementType element[MAXSIZE];
    int front;
    int rear;
}SeqQueue;

  3. 基本操作

    • 初始化:空

      void InitQueue(SeqQueue *Q)
{
    Q->front=0;
    Q->rear=0;
}

    • 循环队列的入队操作（放入尾部）

      int EnterQueue(SeqQueue *Q,QueueElementType x)
{
    if((Q->rear+1)%MAXSIZE==Q->front)   //尾指针追上头指针，已满
        return FALSE;
    Q->element[Q->rear+1]=x;
    Q->rear=(Q->rear+1)%MAXSIZE;
    return TRUE;
}

    • 循环列表的出队操作(删除头元素)

      int DeleteQueue(SeqQueue *Q,QueueElementType *x)
{
    if(Q->front==Q->rear)
        return FALSE;
    *x=Q->element[Q->front];
    Q->front=(Q->front+1)%MAXSIZE;
    return TRUE;
}

    • 还有一种处理假溢出的方法：添加一个标志量tag
